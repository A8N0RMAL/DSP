import numpy as np
import matplotlib.pyplot as plt

def plot_signal(formula, start, end, num_points):
    # Generate x values
    x = np.linspace(start, end, num_points)

    # Evaluate the formula to get y values
    y = eval(formula)

    # Plot the signal
    plt.plot(x, y)
    plt.xlabel('Time')
    plt.ylabel('Amplitude')
    plt.title('Signal')
    plt.grid(True)
    plt.show()

# Prompt the user for the formula
formula = input("Enter the formula for the signal (using 'x' as the variable): ")

# Define the start and end time
start = 0
end = 10

# Define the number of points for the plot
num_points = 1000

# Plot the signal
plot_signal(formula, start, end, num_points)
